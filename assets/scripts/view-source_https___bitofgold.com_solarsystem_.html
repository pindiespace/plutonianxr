var USE_LOGD = false;
var canvas = document.getElementById("renderCanvas");
var engine = new BABYLON.Engine(canvas, true);

var planets = false;
var pscale = 1e-8;
var sunscale = 0.25;

// SETTINGS
var modelscale = 500.0; // scale of planet radiuses
var setsunscale = 1.0; // scale of the sun
var moonscale = 50.0; // scale of moon distance
var timescale = 86400.0 / 24.0; //10 seconds is a day.
var targetplanet = 3; //earth
var autoturn = 'sun';
var rscale = pscale * 1e3 * modelscale;
var skybox = false;

var createRing = function (scene,steps) {
var paths = [];
var step = (2.0 * Math.PI) / (steps -1);
for(var p = 1; p < 3; p += 1) {
var path = [];
for(var i = 0; i < steps; i++) {
var x = p * Math.sin(i * step);
var y = 0;
var z = p * Math.cos(i * step);
path.push(new BABYLON.Vector3(x,y,z));
}
paths.push(path);
}
var ring = BABYLON.Mesh.CreateRibbon("ring", paths, true, false, 0, scene, false, BABYLON.Mesh.BACKSIDE);
return(ring);
};

var createScene = function () {
// This creates a basic Babylon Scene object (non-mesh)
var scene = new BABYLON.Scene(engine);
// This creates and positions a free camera (non-mesh)
scene.camera = new BABYLON.ArcRotateCamera("camera1", -Math.PI / 2.0, Math.PI / 4.0, 20000*rscale, new BABYLON.Vector3(0, 0, 0), scene);
scene.camera.useAutoRotationBehavior = true;
scene.camera.autoRotationBehavior.idleRotationWaitTime = 0.05;
scene.camera.autoRotationBehavior.idleRotationSpinupTime = 0.0;
scene.camera.autoRotationBehavior.idleRotationSpeed = -0.02;
scene.camera.upperRadiusLimit = 40000;
scene.camera.lowerRadiusLimit = 6500*rscale;
scene.camera.maxZ = 220000;
scene.camera.minZ = 0.01;
scene.camera.wheelPrecision = 3 / modelscale;
// This targets the camera to scene origin
scene.camera.setTarget(BABYLON.Vector3.Zero());
// This attaches the camera to the canvas
scene.camera.attachControl(canvas, true);
// This creates a light, aiming 0,1,0 - to the sky (non-mesh)
scene.light = new BABYLON.PointLight("light1", new BABYLON.Vector3(-10000, 0, 0), scene);
scene.light.range = 2000000;
scene.light.position = new BABYLON.Vector3(-10000, 0, 0);
// Default intensity is 1. Let's dim the light a small amount
scene.light.intensity = 1;
planets = window.lagrange.planet_positions.getPositions(new Date(),true);
planets = console.bodies;

// https://nssdc.gsfc.nasa.gov/planetary/factsheet/
var extradata = {
// ,sidereal year ,hours/rotation ,degrees
sun:{ operiod: 1.0, rot:587.28, inclination: 0.0}, //The solar rotation period is 24.47 days at the equator and almost 38 days at the poles. (Wikipedia)
mercury:{ operiod: 0.240846, rot:1407.6, inclination: 0.034},
venus:{ operiod: 0.615, rot:-5832.5, inclination: 177.4},
earth:{ operiod: 1.0, rot:23.9345, inclination: 23.45},
moon:{ operiod: 1.0, rot:655.7, inclination: 6.7 , yaw: -120.0}, //turn the right face to earth
mars:{ operiod: 1.881, rot:24.6, inclination: 25.2},
jupiter:{ operiod: 11.86, rot:9.9, inclination: 3.1},
saturn:{ operiod: 29.46, rot:10.7, inclination: 26.7},
uranus:{ operiod: 84.01, rot:-17.2, inclination: 97.8},
neptune:{ operiod: 164.8, rot:16.1, inclination: 28.3},
pluto:{ operiod: 248.1, rot:-153.3, inclination: 122.5},
halley:{ operiod: 75.316, rot:1.0, inclination: 50.0},
};
for (var p=0;p<planets.length;p++) {
var planet = planets[p];
var ed = extradata[planet.name];
if (ed) {
for (k in ed) {
planet[k] = ed[k];
}
}
}
console.log(console.bodies);

scene.gui = BABYLON.GUI.AdvancedDynamicTexture.CreateFullscreenUI("UI");

for (var p=0;p<planets.length;p++) {
var planet = planets[p];
planet.model = BABYLON.Mesh.CreateSphere("model_"+planet.name, 64, 1.0, scene);
var mat = new BABYLON.StandardMaterial("mat_"+planet.name, scene);
mat.useLogarithmicDepth = USE_LOGD;
planet.model.material = mat;
if (planet.name=="sun") {
mat.emissiveTexture = new BABYLON.Texture("textures/sun.jpg", scene, false, true);
mat.emissiveTexture.level = 3.0;
} else if (planet.name=="mercury") {
mat.diffuseTexture = new BABYLON.Texture("textures/mercury.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.3,0.3,0.3);
} else if (planet.name=="venus") {
mat.diffuseTexture = new BABYLON.Texture("textures/venus.jpg", scene, false, true);
mat.bumpTexture = new BABYLON.Texture("textures/venusnormal.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.2,0.2,0.2);
} else if (planet.name=="earth") {
mat.diffuseTexture = new BABYLON.Texture("textures/earth.jpg", scene, false, true);
mat.emissiveTexture = new BABYLON.Texture("textures/earthnight.jpg", scene, false, true);
mat.specularTexture = new BABYLON.Texture("textures/earthspecular.jpg", scene, false, true);
mat.bumpTexture = new BABYLON.Texture("textures/earthnormal.png", scene, false, true);
planet.model.clouds = BABYLON.Mesh.CreateSphere("clouds_"+planet.name, 64, 1.02, scene);
planet.model.clouds.parent = planet.model;
var cmat = new BABYLON.StandardMaterial("cmat_"+planet.name, scene);
cmat.useLogarithmicDepth = USE_LOGD;
planet.model.clouds.material = cmat;
cmat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
cmat.opacityTexture = new BABYLON.Texture("textures/earthclouds.jpg", scene, false, true);
cmat.opacityTexture.getAlphaFromRGB = true;
cmat.backFaceCulling = false;
cmat.alpha = 0.9;
} else if (planet.name=="moon") {
mat.diffuseTexture = new BABYLON.Texture("textures/moon.jpg", scene, false, true);
mat.bumpTexture = new BABYLON.Texture("textures/moonnormal.jpg", scene, false, true);
mat.bumpTexture.level = 0.5;
mat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
} else if (planet.name=="mars") {
mat.diffuseTexture = new BABYLON.Texture("textures/mars.jpg", scene, false, true);
mat.bumpTexture = new BABYLON.Texture("textures/marsnormal.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
} else if (planet.name=="jupiter") {
mat.diffuseTexture = new BABYLON.Texture("textures/jupiter.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
} else if (planet.name=="saturn") {
mat.diffuseTexture = new BABYLON.Texture("textures/saturn.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
planet.model.ring = createRing(scene,128);
planet.model.ring.scaling.x = 0.75;
planet.model.ring.scaling.y = 0.75;
planet.model.ring.scaling.z = 0.75;
planet.model.ring.parent = planet.model;
var rmat = new BABYLON.StandardMaterial("rmat_"+planet.name, scene);
rmat.useLogarithmicDepth = USE_LOGD;
planet.model.ring.material = rmat;
rmat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
rmat.emissiveTexture = new BABYLON.Texture("textures/saturnring.jpg", scene, false, true);
rmat.emissiveTexture.level = 2.0;
rmat.emissiveTexture.vScale = 0.5;
rmat.opacityTexture = new BABYLON.Texture("textures/saturnring.jpg", scene, false, true);
rmat.opacityTexture.vScale = 2.0;
rmat.opacityTexture.getAlphaFromRGB = true;
rmat.backFaceCulling = false;
rmat.alpha = 0.5;
} else if (planet.name=="uranus") {
mat.diffuseTexture = new BABYLON.Texture("textures/uranus.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
} else if (planet.name=="neptune") {
mat.diffuseTexture = new BABYLON.Texture("textures/neptune.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.1,0.1,0.1);
} else if (planet.name=="pluto") {
mat.diffuseTexture = new BABYLON.Texture("textures/pluto.jpg", scene, false, true);
mat.bumpTexture = new BABYLON.Texture("textures/plutonormal.jpg", scene, false, true);
mat.specularColor = new BABYLON.Color3(0.3,0.3,0.3);
}

var size = 0.8;
planet.model.axisY = BABYLON.Mesh.CreateLines("axisY_"+planet.name, [
new BABYLON.Vector3(0, -size, 0), new BABYLON.Vector3(0, size, 0)
], scene);
planet.model.axisY.color = new BABYLON.Color3(1, 1, 1);
planet.model.axisY.parent = planet.model;
planet.model.axisY.alpha = 0.5;


if (p==0) { //sun
var r = planet.radius * rscale * sunscale;
} else {
var r = planet.radius * rscale;
}
planet.model.scaling.x = r;
planet.model.scaling.y = r;
planet.model.scaling.z = r;
planet.model.radius = r;

planet.model.position.x = planet.position.y;
planet.model.position.y = -planet.position.z;
planet.model.position.z = -planet.position.x;
var distance = planet.model.position.length();
planet.model.position.scaleInPlace(pscale);
console.log(planet.name+ ' r = '+planet.radius.toFixed(0)+'km, distance = '+(distance/149597870700).toFixed(2)+'AU');

planet.model.baseRotationQuaternion = BABYLON.Quaternion.RotationYawPitchRoll(0.0, 0.0, 3.14159264 - (planet.inclination ? planet.inclination * Math.PI / 360: 0.0));
planet.model.rotationQuaternion = new BABYLON.Quaternion();
planet.model.alpha = 0.0;

var rect1 = new BABYLON.GUI.Rectangle();
rect1.width = "150px";
rect1.height = "40px";
rect1.color = "rgba(255,255,255,0.37)";
rect1.thickness = 0;
rect1.background = "transparent";
scene.gui.addControl(rect1);

var label = new BABYLON.GUI.TextBlock();
label.text = planet.title;
rect1.addControl(label);

rect1.linkWithMesh(planet.model);
rect1.linkOffsetY = -40;

}

// Skybox
scene.skybox = BABYLON.Mesh.CreateSphere("skyBox", 32, 200000.0, scene);
scene.skybox.rotation.x = -0.5;
scene.skybox.rotation.y = 1.7;
scene.skybox.rotation.z = 4.1;
//scene.skybox.infiniteDistance = true;
var skyboxMaterial = new BABYLON.StandardMaterial("skyBox", scene);
skyboxMaterial.useLogarithmicDepth = USE_LOGD;
scene.skybox.material = skyboxMaterial;
skyboxMaterial.backFaceCulling = false;
skyboxMaterial.emissiveTexture = new BABYLON.Texture("textures/sky.jpg", scene, false, true);
skyboxMaterial.disableLighting = true;
return scene;
};
var scene = createScene();
var starttime = Date.now();

engine.runRenderLoop(function () {
rscale = pscale * 1e3 * modelscale;
sunscale = 0.02 * setsunscale;

var timenow = (Date.now() - starttime) / 1000.0 * timescale;
var ppos = window.lagrange.planet_positions.getPositions(timenow * 1000.0,true);
for (var p=0;p<planets.length;p++) {
var planet = planets[p];

if (p==0) { //sun
var r = planet.radius * rscale * sunscale;
} else {
var r = planet.radius * rscale;
}
planet.model.scaling.x = r;
planet.model.scaling.y = r;
planet.model.scaling.z = r;
planet.model.radius = r;

planet.model.position.x = ppos[p].position.y;
planet.model.position.y = -ppos[p].position.z;
planet.model.position.z = -ppos[p].position.x;
planet.model.position.scaleInPlace(pscale);
planet.model.alpha = timenow / 3600.0 / planet.rot * Math.PI * 2.0;
if (planet.yaw) { // moon
planet.model.alpha += planet.yaw * Math.PI / 360;
}
planet.model.rotationQuaternion.copyFrom(planet.model.baseRotationQuaternion);
planet.model.rotationQuaternion.multiplyInPlace(BABYLON.Quaternion.RotationYawPitchRoll(planet.model.alpha, 0.0, 0.0)); //because upside down (textures)
if (p==3) { //earth
planet.model.clouds.rotation.y = timenow / 3600.0 / planet.rot * 0.15 * Math.PI * 2.0;
}
if (p==11) { //moon
planet.model.position.subtractInPlace(planets[3].model.position);
planet.model.position.scaleInPlace(moonscale);
planet.model.position.addInPlace(planets[3].model.position);
}
}

scene.light.position.copyFrom(planets[0].model.position);

if (autoturn=='no') {
scene.camera.autoRotationBehavior.idleRotationSpeed = 0.0;
} else if (autoturn=='sun') {
scene.camera.autoRotationBehavior.idleRotationSpeed = -2.0 * Math.PI / (planets[targetplanet].operiod * 86400.0 * 365.4) * timescale;
} else if (autoturn=='moon') {
scene.camera.autoRotationBehavior.idleRotationSpeed = -2.0 * Math.PI / (planets[11].rot * 3600.0) * timescale;
} else if (autoturn=='target') {
scene.camera.autoRotationBehavior.idleRotationSpeed = -2.0 * Math.PI / (planets[targetplanet].rot * 3600.0) * timescale;
}
scene.camera.target.copyFrom(planets[targetplanet].model.position);
scene.camera.minZ = 0.01 * modelscale;
scene.camera.upperRadiusLimit = 40000;
scene.camera.lowerRadiusLimit = planets[targetplanet].radius*1.02*rscale;
if (targetplanet==0) { scene.camera.lowerRadiusLimit *= sunscale;}
if (scene.camera.lowerRadiusLimit<scene.camera.minZ*1.7) {scene.camera.lowerRadiusLimit = scene.camera.minZ*1.7;}
scene.camera.wheelPrecision = 6000.0 / modelscale * (1 / (scene.camera.radius / scene.camera.lowerRadiusLimit));

scene.skybox.position.copyFrom(scene.camera.globalPosition);

scene.render();
});

// Resize
window.addEventListener("resize", function () {
engine.resize();
});
